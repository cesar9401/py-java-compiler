  	public int getResta() {
		int value = 10;

        	value = value - 1;
          	if(value == 4) {
            	break;
            }
        }
      	return this.value1 - this.value2;
    }

      	println(this.value1, this.value2);
      	for(int i = 0; i < 10; i = i + 1) {
          	if(i == 2) {
            	continue;
            } else if(i == 5) {
            	break;
            }
        	println("Iteracion:", i);
        }
    }
  
  	public int factorial(int n) {
    	if(n < 0) {
        	return -1;
        } else if(n <= 1) {
        	return 1;
        } else {
        	return n * factorial(n - 1);
        }
    }
  	
  	public int ackerman(int m, int n) {
    	if(m == 0) {
        	return n + 1;
        } else if(m > 0 && n == 0) {
        	return ackerman(m - 1, 1);
        } else {
          	int value = ackerman(m, n - 1);
        	return ackerman(m  - 1, value);
        }
    }
  
}
